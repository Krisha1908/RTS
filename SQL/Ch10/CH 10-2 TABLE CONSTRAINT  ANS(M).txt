CREATE BELOW TABLES WITH DEFAULT AND CONSTRAINTS REQUIRED :

1. CREATE TABLE TO STORE BELOW INFO.
   FACULTIES : FACULTY_NO, FACNM, JOINDATE, SALARY

=> CREATE TABLE FACULTIES (FACULTY_NO NUMBER CHECK(FACULTY_NO BETWEEN 1 AND 100) PRIMARY KEY, 
			   FACNM VARCHAR (40), JOINDATE DATE  DEFAULT SYSDATE, 
			   SALARY NUMBER(10) NOT NULL );

2. CREATE TABLE TO STORE BELOW INFO.
   SUBJECTS : SUBJECT_NO, SUBNM, DURATION_IN_DAYS

==>  CREATE TABLE SUBJECTS (SUBJECT_NO NUMBER CHECK (SUBJECT_NO BETWEEN 1 AND 100) PRIMARY KEY, 
			    SUBNM VARCHAR (40) NOT NULL UNIQUE,
                            DURATION_IN_DAYS INTERVAL DAY TO SECOND CHECK(DURATION_IN_DAYS > '0 00:00:00.000000'));

3. CREATE TABLE TO STORE BELOW INFO.
   FACULTY_SUBJECT : FACULTY_NO, SUBJECT_NO

=>CREATE TABLE FACULTY_SUBJECT (FACULTY_NO NUMBER REFERENCES FACULTIES, SUBJECT_NO NUMBER REFERENCES SUBJECTS);

4. CREATE TABLE TO STORE BELOW INFO.
   COURSE : COURSE_NO, COURSE_NAME, DURATION, GENERAL_CLUB, TOTAL_FEES

=> CREATE TABLE COURSE (COURSE_NO NUMBER(2) CHECK (COURSE_NO BETWEEN 1 AND 10) PRIMARY KEY,
                        COURSE_NAME VARCHAR(40) UNIQUE NOT NULL,
                        DURATION  INTERVAL DAY TO  SECOND CHECK(DURATION > '00 00:00:00.000') ,
                        GENERAL_CLUB VARCHAR (8)  NOT NULL,
                        TOTAL_FEES NUMBER (8,2) NOT NULL);
